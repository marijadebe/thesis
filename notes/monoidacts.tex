\section{Monoids and acts}
\begin{definition}
    A 3-tuple $\mathcal{S} = (S,\star,e)$ where $S$ is a set and $\star : S\times S \to S$, $e : 1 \to S$ are 
    maps is called a \textbf{monoid}, if the following two conditions hold
    \begin{enumerate}
        \item $\forall x,y,z\in S : (x\star y)\star z = x\star(y\star z)$
        \item $\forall x\in S : x\star e = x = e \star x$
    \end{enumerate}
\end{definition}
\begin{example}
    Let $X$ be a nonempty set. Denote by $T(X)$ the set of all maps
    $X\to X$.\\ Then $\mathcal{T}(X) = (T(X),\circ,\text{id}_X) $, where $\circ$ is a 
    composition of maps and $\text{id}_X$ the identity map, is a monoid. We will
    call $\mathcal{T}(X)$ the transformation monoid of $X$.\par
    Let $\mathcal{S} = (S,\star,e)$ be a monoid, we define the opposite monoid $\opp{\mathcal{S}}=(S,\star_{\text{op}}, e)$ where
    \[\forall x,y\in S: x\star_{\text{op}} y = y\star x\]
\end{example}
\begin{definition}
    Let $\mathcal{S}$ be a monoid and $X$ a nonempty set. A tuple $(X,\mu)$, where $\mu : S\times X \to X$ is called 
    a \textbf{left $\mathcal{S}$-act} if the map $\mu$ satisfies the following
    \begin{enumerate}
        \item $\forall r,s \in S\,\forall x \in X : \mu(rs, x) = \mu(r,\mu(s,x))$
        \item $\forall x \in S : \mu(e,x) = x$
    \end{enumerate}
    we will denote a left $\mathcal{S}$-act as ${}_{\mathcal{S}}X$. Similarly we
    define a right $\mathcal{S}$-act and use the notation $X_\mathcal{S}$.
\end{definition}
\begin{remark}
    When we use the word $\mathcal{S}$-act we will mean the left $\mathcal{S}$-act, unless 
    stated otherwise.
\end{remark}
\begin{definition}
    Let $X$ be a set and $\mathcal{S}$ a monoid. We define the \textbf{free $\mathcal{S}$-act on X} to be 
    the tuple  
    \[(S\times X,\cdot : S\times (S\times X) \to S\times X)\]
    where $s_1\cdot (s,x) = (s_1s,x)$.
\end{definition}
It is an act, since $e\cdot(s,x) = (es,x) = (s,x)$ and $s_1s_2\cdot(s,x) = 
((s_1s_2)s,x) = (s_1(s_2s),x) = s_1\cdot(s_2s,x) = s_1\cdot(s_2\cdot(s,x))$.
\begin{definition}
    Given $\act{S}{X}$ and $\act{S}{Y}$ we call $f: X \to Y$ an
    \textbf{act homomorphism}, if 
    \[
        \forall s\in S\,\forall x \in X : f(s\cdot x) = s\cdot f(x)
    \]
\end{definition}
\begin{proposition}
    Acts with their morphisms form a category, which we will denote $\cat{$\mathcal{S}$-act}$.
\end{proposition}
