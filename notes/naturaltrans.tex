\subsection{Natural transformations}
\begin{definition}
    Let $F,G: \catc \to \catd$ be a pair of functors. A $\textbf{natural transformation}$ $\eta : F \to G$ is a 
    collection of morphisms in $\catd$ indexed by the objects of $\catc$. Explicitly $\eta = \{\eta_A : F(A)\to G(A) \mid A \in\ob{\catc}\}$, such that
    for all objects $A,B\in\catc$ and morphisms $f : A\to B$ the following square diagram commutes
\[\begin{tikzcd}
	{F(A)} && {F(B)} \\
	\\
	{G(A)} && {G(B)}
	\arrow["{F(f)}", from=1-1, to=1-3]
	\arrow["{\eta_A}"{description}, from=1-1, to=3-1]
	\arrow["{\eta_B}"{description}, from=1-3, to=3-3]
	\arrow["{G(f)}", from=3-1, to=3-3]
\end{tikzcd}\]
For each $A\in\catc$ we call $\eta_A$ \textbf{the component of $\eta$ at $A$}.
A natural transformation $\eta$ is a \textbf{natural isomorphism} if for all $A\in\ob{\catc}$ the morphism $\eta_A$ is an isomorphism.
\end{definition}
\begin{definition}
    Given two natural transformations $\alpha:F\to G$ and $\beta:G\to H$ as such 
\[\begin{tikzcd}
	\catc && \catd
	\arrow[""{name=0, anchor=center, inner sep=0}, "F", shift left, curve={height=-18pt}, from=1-1, to=1-3]
	\arrow[""{name=1, anchor=center, inner sep=0}, "H"', shift right, curve={height=18pt}, from=1-1, to=1-3]
	\arrow[""{name=2, anchor=center, inner sep=0}, "G"{description}, from=1-1, to=1-3]
	\arrow["\alpha", shorten <=3pt, shorten >=3pt, Rightarrow, from=0, to=2]
	\arrow["\beta", shorten <=3pt, shorten >=3pt, Rightarrow, from=2, to=1]
\end{tikzcd}\]
    We define their \textbf{vertical composite} $\beta\circ\alpha$ componentwise by taking
    $\forall A\in\ob{\catc} : (\beta\circ\alpha)_A = \beta_A\circ\alpha_A$.
\end{definition}
\begin{proposition}
    The horizontal composite $\beta\circ\alpha: F\to H$ is a natural transformation. 
\end{proposition}
\begin{proof}
    We need to show that the square diagram
\[\begin{tikzcd}
	{F(A)} && {H(A)} \\
	\\
	{F(B)} && {H(B)}
	\arrow["{(\beta\circ\alpha)_A}"{description}, from=1-1, to=1-3]
	\arrow["{F(f)}", from=1-1, to=3-1]
	\arrow["{H(f)}"{description}, from=1-3, to=3-3]
	\arrow["{(\beta\circ\alpha)_B}"{description}, from=3-1, to=3-3]
\end{tikzcd}\]
    commutes. From the definition of $(\beta\circ\alpha)$ we have
    \[\begin{tikzcd}
	{F(A)} && {G(A)} && {H(A)} \\
	\\
	{F(B)} && {G(B)} && {H(B)}
	\arrow["{\alpha_A}"{description}, from=1-1, to=1-3]
	\arrow["{F(f)}", from=1-1, to=3-1]
	\arrow["{\beta_A}"{description}, from=1-3, to=1-5]
	\arrow["{G(f)}"{description}, from=1-3, to=3-3]
	\arrow["{H(f)}"{description}, from=1-5, to=3-5]
	\arrow["{\alpha_B}"{description}, from=3-1, to=3-3]
	\arrow["{\beta_B}"{description}, from=3-3, to=3-5]
\end{tikzcd}\]
    and from the definition of natural transformation it follows that the left and right squares
    commute, so the whole diagram commutes as well.
\end{proof}
\begin{definition}
    Let $1_F : F\to F$ be defined componentwise as $(1_F)_A = 1_{F(A)}$, then $1_F$
    is a natural transformation which we will call the identity.
\end{definition}
\begin{definition}
    Given any two categories $\catc,\catd$ we define a \textbf{functor category} $\funcat{\catc}{\catd}$
    whose objects are functors from $\catc$ to $\catd$ and whose morphisms are
    natural transformations. We define the composition in $\funcat{\catc}{\catd}$ as horizontal
    composition and we let the identity morphisms be the identity natural transformations.
\end{definition}
\begin{proposition}
    $\funcat{\catc}{\catd}$ is a category.
\end{proposition}
\begin{proof}[Proof]
    Choose $A\in\ob{\catc}$, then 
    \begin{equation*}
        (\gamma\circ(\beta\circ\alpha))_A = \gamma_A\circ(\beta\circ\alpha)_A
        = \gamma_A\circ(\beta_A\circ\alpha_A) = (\gamma_A\circ\beta_A)\circ\alpha_A = 
        (\gamma\circ\beta)_A\circ\alpha_A = ((\gamma\circ\beta)\circ\alpha)_A
    \end{equation*}
    \begin{equation*}
        (\alpha\circ 1_F)_A = \alpha \circ 1_{F_(A)} = \alpha_A \hspace{20pt} 
        (1_F\circ\beta)_A = 1_{F(A)}\circ\beta_A = \beta_A
    \end{equation*}
\end{proof}
\begin{definition}
    Given any four functors and two natural transformations like so
    \[\begin{tikzcd}
	\catc && \catd && \cate
	\arrow[""{name=0, anchor=center, inner sep=0}, "{F_1}", curve={height=-12pt}, from=1-1, to=1-3]
	\arrow[""{name=1, anchor=center, inner sep=0}, "{G_1}"', curve={height=12pt}, from=1-1, to=1-3]
	\arrow[""{name=2, anchor=center, inner sep=0}, "{F_2}", curve={height=-12pt}, from=1-3, to=1-5]
	\arrow[""{name=3, anchor=center, inner sep=0}, "{G_2}"', curve={height=12pt}, from=1-3, to=1-5]
	\arrow["\alpha", shorten <=3pt, shorten >=3pt, Rightarrow, from=0, to=1]
	\arrow["\beta", shorten <=3pt, shorten >=3pt, Rightarrow, from=2, to=3]
\end{tikzcd}\]
    we define \textbf{the horizontal composite} $\beta\ast\alpha:F_2\circ F_1 \to G_2\circ G_1$ by
    \[
        \forall A\in\catc : (\beta\ast\alpha)_A = \beta_{G_1(A)}\circ F_2(\alpha_A)
    \]
\end{definition}
\begin{remark}
    Notice that $\beta_{G_1(A)}\circ F_2(\alpha_A)=G_2(\alpha_A)\circ\beta_{F_1(A)}$ by 
    the naturality of $\beta$ at $F_1(A)$ and $G_1(A)$.
\end{remark}
\begin{theorem}[Middle-four interchange law]
    Consider the following diagram
\[\begin{tikzcd}
	\catc && \catd && \cate
	\arrow[""{name=0, anchor=center, inner sep=0}, "{F_1}", curve={height=-24pt}, from=1-1, to=1-3]
	\arrow[""{name=1, anchor=center, inner sep=0}, "{H_1}"', curve={height=24pt}, from=1-1, to=1-3]
	\arrow[""{name=2, anchor=center, inner sep=0}, "{G_1}"{description}, from=1-1, to=1-3]
	\arrow[""{name=3, anchor=center, inner sep=0}, "{F_1}", curve={height=-24pt}, from=1-3, to=1-5]
	\arrow[""{name=4, anchor=center, inner sep=0}, "{H_2}"', curve={height=24pt}, from=1-3, to=1-5]
	\arrow[""{name=5, anchor=center, inner sep=0}, "{G_2}"{description}, from=1-3, to=1-5]
	\arrow["{\alpha_1}", shorten <=3pt, shorten >=3pt, Rightarrow, from=0, to=2]
	\arrow["{\alpha_2}", shorten <=3pt, shorten >=3pt, Rightarrow, from=2, to=1]
	\arrow["{\beta_1}", shorten <=3pt, shorten >=3pt, Rightarrow, from=3, to=5]
	\arrow["{\beta_2}", shorten <=3pt, shorten >=3pt, Rightarrow, from=5, to=4]
\end{tikzcd}
\]
then $(\beta_2\circ\beta_1)\ast(\alpha_2\circ\alpha_1) = (\beta_2\ast\alpha_2)\circ(\beta_1\ast\alpha_1)$.
\end{theorem}

\begin{definition}
	Given a diagram like so
	\[\begin{tikzcd}
	\catc && \catd && \cate
	\arrow[""{name=0, anchor=center, inner sep=0}, "G"', curve={height=12pt}, from=1-1, to=1-3]
	\arrow[""{name=1, anchor=center, inner sep=0}, "F", curve={height=-12pt}, from=1-1, to=1-3]
	\arrow["H", from=1-3, to=1-5]
	\arrow["\alpha", shorten <=3pt, shorten >=3pt, Rightarrow, from=1, to=0]
\end{tikzcd}\]
	we define the \textbf{whiskering of $H$ and $\alpha$} as
	a natural transformation $H\alpha: HF \to HG$ where
	\[
		\forall A\in\ob{\catc} : (H\alpha)_A = H(\alpha_A)
	\]  
\end{definition}
\begin{definition}
	Given a diagram 
	\[\begin{tikzcd}
	\catc && \catd && \cate
	\arrow["F", from=1-1, to=1-3]
	\arrow[""{name=0, anchor=center, inner sep=0}, "H"', curve={height=12pt}, from=1-3, to=1-5]
	\arrow[""{name=1, anchor=center, inner sep=0}, "G", curve={height=-12pt}, from=1-3, to=1-5]
	\arrow["\beta", shorten <=3pt, shorten >=3pt, Rightarrow, from=1, to=0]
	\end{tikzcd}\]
	we define the \textbf{whiskering of $\beta$ and $F$} as a 
	natural transformation $\beta F: GF \to HF$ where
	\[
		\forall A\in\ob{\catc} : (\beta F)_A = \beta_{F(A)}
	\]
\end{definition}
\begin{remark}
	\[
		H\alpha = 1_H\ast\alpha \hspace{20pt} \beta F = \beta\ast 1_F
	\]
\end{remark}
\begin{definition}
	Given two functors $\begin{tikzcd}
		\catc & \catd
		\arrow["F", shift left, from=1-1, to=1-2]
		\arrow["G", shift left, from=1-2, to=1-1]
	\end{tikzcd}$ then the 
	\textbf{adjunction} $F\dashv G$ is given by a pair
	of natural transformations $\eta : 1_\catc \to GF$ called
	the \textbf{unit} and $\epsilon: FG \to 1_\catd$ called
	the \textbf{counit} such that the following diagrams commute
	(in their respective functor categories) 
\[\begin{tikzcd}
	F && FGF && G && GFG \\
	\\
	&& F &&&& G
	\arrow["{F\eta}"{description}, from=1-1, to=1-3]
	\arrow["{1_F}", from=1-1, to=3-3]
	\arrow["{\epsilon F}"{description}, from=1-3, to=3-3]
	\arrow["{\eta G}"{description}, from=1-5, to=1-7]
	\arrow["{1_F}", from=1-5, to=3-7]
	\arrow["{G\epsilon}"{description}, from=1-7, to=3-7]
\end{tikzcd}\]

\end{definition}
\begin{definition}
	Let $F : \catc \to \catc $ be an endofunctor, $\eta : 1_{\catc} \to F$ and 
	$\mu : F^2 \to F$ a pair of natural transformations. We then call the triple
	$(F,\mu,\eta)$ a \textbf{monad}, if the following two diagrams commute
	\[\begin{tikzcd}
	F & {F^2} & F && {F^3} & {F^2} \\
	& F &&& {F^2} & F
	\arrow["{F\eta}", from=1-1, to=1-2]
	\arrow[Rightarrow, no head, from=1-1, to=2-2]
	\arrow["\mu", from=1-2, to=2-2]
	\arrow["{\eta F}"', from=1-3, to=1-2]
	\arrow[Rightarrow, no head, from=1-3, to=2-2]
	\arrow["{F\mu}", from=1-5, to=1-6]
	\arrow["{\mu F}"', from=1-5, to=2-5]
	\arrow["\mu", from=1-6, to=2-6]
	\arrow["\mu"', from=2-5, to=2-6]
\end{tikzcd}\]
\end{definition}
\begin{proposition}
    Functor $F:\catc \to \catd$ is an equivalence of categories if and only if 
    $F$ is full, faithful and essentialy surjective on objects.
\end{proposition}
\begin{proof}[Proof]
    \begin{enumerate}
        \item $F$ equivalence $\implies F$ faithful\par
        Choose $f,g : A\to B$ and suppose that $F(f)=F(g)$, there exists a 
        functor $G: \catd \to \catc$ and a natural isomorphism $\eta: G\circ F \to 1_\catc$.\par
        Consider the following diagram 
\[\begin{tikzcd}
	A \\
	{G(F(A))} && {G(F(B))} && B \\
	A
	\arrow["f", from=1-1, to=2-5]
	\arrow["{\eta_A}", from=2-1, to=1-1]
	\arrow["{G(F(f))=G(F(g))}"', from=2-1, to=2-3]
	\arrow["{\eta_A}"', from=2-1, to=3-1]
	\arrow["{\eta_B}"{description}, from=2-3, to=2-5]
	\arrow["g"', from=3-1, to=2-5]
\end{tikzcd}\]
    both the upper and lower squares commute by naturality, hence
    $f\circ\eta_A = g\circ\eta_A \implies f=g$ since $\eta_A$ is 
    an isomorphism.
    \item $F$ equivalence $\implies F$ full\par
    Let $g: F(A)\to F(B)$ and define $f = \eta_B\circ G(g)\circ\eta_A^{-1}$
    then $f: A\to B$. Since $G$ is an equivalence of categories it suffices
    to show that $GF(f)=G(g)$ by $1$. From naturality we have that
    \[\begin{tikzcd}
	{GF(A)} && {GF(B)} \\
	A && B
	\arrow["{GF(f)}", from=1-1, to=1-3]
	\arrow["{\eta_A}", from=1-1, to=2-1]
	\arrow["f", from=2-1, to=2-3]
	\arrow["{\eta_B}", from=1-3, to=2-3]
\end{tikzcd}\]
    commutes and therefore
    \[
        GF(f) = \eta_B^{-1}\circ f \circ\eta_A = G(g) \implies F(f)=g
    \]
	\item $F$ equivalence $\implies F$ essentialy surjective on objects\par 
    Choose an object $B\in\ob{\catd}$, then we have $\mu_B : F(G(B)) \to B$ an isomorphism.

	\item $F$ fully faithful and essentialy surjective on objects $\implies$ $F$ is an equivalence
	\par
	For $B\in\ob{\catd}$ choose $A\in\ob{\catc}$ such that $F(A)\cong B$ fix any such
	isomorphism and let 
	$G(B)=A$. Given any morphism $g: B \to C$ in $\catd$ we have
	\[\begin{tikzcd}
	{FG(B)} & B & C & {FG(C)}
	\arrow["{\mu_1}", shift left, from=1-1, to=1-2]
	\arrow["{\mu_1^{-1}}", shift left, from=1-2, to=1-1]
	\arrow["g", from=1-2, to=1-3]
	\arrow["{\mu_2}", shift left=2, from=1-3, to=1-4]
	\arrow["{\mu_2^{-1}}", shift left, from=1-4, to=1-3]
\end{tikzcd}\]
	by full faithfulness of $F$ there exists a unique morphism
	$f: G(B) \to G(C)$ such that $F(f) = \mu_2\circ g \circ \mu_1$. We let
	$G(g) = f$. We will now prove, that $F(G(g_2)\circ G(g_1)) = FG(g_2\circ g_1)$. Then
	it will follow by faithfulness, that $G$ respects composition.
	\[\begin{tikzcd}
	B & C & D \\
	{FG(B)} & {FG(C)} & {FG(D)}
	\arrow["{g_1}", from=1-1, to=1-2]
	\arrow["{\mu_B^{-1}}", shift left, from=1-1, to=2-1]
	\arrow["{g_2}", from=1-2, to=1-3]
	\arrow["{\mu_C}", shift left=2, from=1-2, to=2-2]
	\arrow["{\mu_D}", shift left, from=1-3, to=2-3]
	\arrow["{\mu_B}", shift left, from=2-1, to=1-1]
	\arrow["{\mu_C^{-1}}", shift left, from=2-2, to=1-2]
	\arrow["{\mu_D^{-1}}", shift left, from=2-3, to=1-3]
\end{tikzcd}\]
	We have $F(G(g_2) G(g_1))=FG(g_2) FG(g_1)
	= \mu_3 g_2 \mu_2^{-1} \mu_2 g_1 \mu_1 = \mu_3 g_2 g_1 \mu_1
	= FG(g_2\circ g_1)$ and $FG(1_B) = 1_{FG(B)} = F(1_{G(B)})$.
	Now we have to check, that $FGB\cong B$ naturally in $B$.

\end{enumerate}
\end{proof}
