\section{Products}

\begin{definition}
    Let $X,Y\in\ob{\catc}$, then a \textbf{product of $X$ and $Y$} is a triple $(U,p_X : U \to X, p_Y : U \to Y)$ such that
    for any other triple $(U', q_X : U' \to X, q_Y: U' \to Y)$ there exists a unique morphism denoted by $(q_X,q_Y)$
    such that the following diagram commutes
    \[\begin{tikzcd}
	& {U'} \\
	\\
	{X_1} & U & Y
	\arrow["{q_{X}}"', from=1-2, to=3-1]
	\arrow["{(q_X,q_Y)}"{description}, from=1-2, to=3-2]
	\arrow["{q_{Y}}", from=1-2, to=3-3]
	\arrow["{p_{X}}", from=3-2, to=3-1]
	\arrow["{p_{Y}}"', from=3-2, to=3-3]
    \end{tikzcd}\]
\end{definition}
\begin{lemma}
    \label{uniqueselfmapLemma}
    Let $(U,p_X,p_Y)$ be a product of $X$ and $Y$, then $1_U$ is the unique morphism which makes
    the following diagram commute 
    \[\begin{tikzcd}
	& U \\
	X && Y \\
	& U
	\arrow["{p_X}"', from=1-2, to=2-1]
	\arrow["{p_Y}", from=1-2, to=2-3]
	\arrow["{1_U}"{description}, from=1-2, to=3-2]
	\arrow["{p_X}", from=3-2, to=2-1]
	\arrow["{p_Y}"', from=3-2, to=2-3]
    \end{tikzcd}\]
\end{lemma}
\begin{proof}[Proof]
    Follows directly from the universal property.
\end{proof}
\begin{lemma}
    \label{uniqueproductLemma}
    Let $(X\times Y, p_X, p_Y)$ and $(X\times' Y, q_X, q_Y)$ be products of $X$ and $Y$, then there exists
    a unique isomorphism, which makes the following diagram commute
    \[\begin{tikzcd}
        & {X\times Y} \\
        \\
        {X_1} & {X\times' Y} & Y
        \arrow["{{p_{X}}}"', from=1-2, to=3-1]
        \arrow["{\exists !\newline \simeq}"{description}, from=1-2, to=3-2]
        \arrow["{{p_{Y}}}", from=1-2, to=3-3]
        \arrow["{{q_{X}}}", from=3-2, to=3-1]
        \arrow["{{q_{Y}}}"', from=3-2, to=3-3]
    \end{tikzcd}\]
    We will denote a representant of this isomorphism class as $X\times Y$.
\end{lemma}
\begin{proof}[Proof]
    Since $X\times Y$ is a product, there exists a unique morphism $(p_X,p_Y): X\times Y \to X\times' Y$, which
    makes the triangles commute. Since 
    $X\times' Y $ is a product, there exists a unique morphism $(q_X, q_Y): X\times' Y \to X\times Y$, which makes the 
    triangles commute. By composing them we get 
    \[\begin{tikzcd}
	&& {X_1} \\
	{X\times Y} && {X\times' Y} && {X\times Y} \\
	&& Y
	\arrow["{{p_{X}}}", from=2-1, to=1-3]
	\arrow["{(p_X,p_Y)}"', shift right, from=2-1, to=2-3]
	\arrow["{{p_{Y}}}"', from=2-1, to=3-3]
	\arrow["{{q_{X}}}"', from=2-3, to=1-3]
	\arrow["{(q_X,q_Y)}", from=2-3, to=2-5]
	\arrow["{{q_{Y}}}", from=2-3, to=3-3]
	\arrow["{p_X}"', from=2-5, to=1-3]
	\arrow["{p_Y}"', from=2-5, to=3-3]
    \end{tikzcd}\]
    since the small triangles commute the two big triangles commute as well, by lemma \ref{uniqueselfLemma} we get
    that $(q_X,q_Y)\circ(p_X,p_Y) = 1_{X\times Y}$. Since the situation is completely symmetrical it holds that 
    $(p_X,p_Y)\circ (q_X,q_Y) = 1_{X\times' Y}$.
\end{proof}
\begin{lemma}
    \label{equalonprojLemma}
    Consider the following commutative diagram 
    \[\begin{tikzcd}
	& U \\
	X & {X\times Y} & Y
	\arrow["{p_X\circ g = p_X\circ f}"', from=1-2, to=2-1]
	\arrow["f", shift left, from=1-2, to=2-2]
	\arrow["g"', shift right, from=1-2, to=2-2]
	\arrow["{p_Y\circ g =p_Y\circ f}", from=1-2, to=2-3]
	\arrow["{p_X}", from=2-2, to=2-1]
	\arrow["{p_Y}"', from=2-2, to=2-3]
\end{tikzcd}\]
    it follows that two morphisms to the product object are equal if any only if they are equal on projections.
\end{lemma}
\begin{proof}[Proof]
    From the universal property of products it follows, that there exists a unique morphism $U\to X\times Y$
    that makes the triangles in the statement commute, but $f$ and $g$ already are such morphisms, therefore
    $f=g$
\end{proof}
\begin{definition}
    Let $f: X\to X'$ and $g: Y\to Y'$, then by $f\times g: X\times Y \to X'\times Y'$ we will mean
    the unique morphism, which makes the following diagram commute
    \[\begin{tikzcd}
	& {X\times Y} \\
	X && Y \\
	{X'} & {X'\times Y'} & {Y'}
	\arrow[from=1-2, to=2-1]
	\arrow[from=1-2, to=2-3]
	\arrow["{f\times g}"{description}, from=1-2, to=3-2]
	\arrow["f", from=2-1, to=3-1]
	\arrow["g", from=2-3, to=3-3]
	\arrow["{p_X}", from=3-2, to=3-1]
	\arrow["{q_X}"', from=3-2, to=3-3]
\end{tikzcd}\]
\end{definition}
\begin{definition}
    In category $\catc$ with binary products we define the \textbf{product bifunctor} $-\times -: \catc\times\catc \to \catc$ by
    \[(X,Y) \mapsto X\times Y\]
    \[(f: X \to X', g: Y \to Y') \mapsto f\times g\]
    We will now show that it is a functor.
\end{definition}
\begin{proof}[Proof]
    Consider the following diagram
    \[\begin{tikzcd}
	{X_1} && {X_1\times Y_1} && {Y_1} \\
	\\
	{X_2} && {X_1\times Y_2} && {Y_2} \\
	\\
	{X_3} && {X_3\times Y_3} && {Y_3}
	\arrow["{f_1}"{description}, from=1-1, to=3-1]
	\arrow["{p_{X_1}}"{description}, from=1-3, to=1-1]
	\arrow["{p_{Y_1}}"{description}, from=1-3, to=1-5]
	\arrow["{f_1\times g_1}"{description}, from=1-3, to=3-3]
	\arrow["{(f_2\circ f_1)\times(g_2\circ g_1)}"{description, pos=0.4}, curve={height=-30pt}, from=1-3, to=5-3]
	\arrow["{g_1}"{description}, from=1-5, to=3-5]
	\arrow["{f_2}"{description}, from=3-1, to=5-1]
	\arrow["{p_{X_2}}"{description}, from=3-3, to=3-1]
	\arrow["{p_{Y_2}}"{description}, from=3-3, to=3-5]
	\arrow["{f_2\times g_2}"{description}, from=3-3, to=5-3]
	\arrow["{g_2}"{description}, from=3-5, to=5-5]
	\arrow["{p_{X_3}}"{description}, from=5-3, to=5-1]
	\arrow["{p_{Y_3}}"{description}, from=5-3, to=5-5]
\end{tikzcd}\]
    All of the small squares commute by definition of product morphism. We have to show that the inner triangle commutes. By 
    lemma \ref{equalonprojLemma} we have to show that the morphisms postcomposed with projections $p_{X_3},p_{Y_3}$ are equal,
    which can be shown by a diagram chase.\par
    It remains to prove that $1_X\times 1_Y = 1_{X\times Y}$. We have that the following diagram commutes
    \[\begin{tikzcd}
	&& {X\times Y} \\
	\\
	X && {X\times Y} && Y
	\arrow["{1_X\circ p_X}"', from=1-3, to=3-1]
	\arrow["{1_X\times 1_Y}"', shift right, from=1-3, to=3-3]
	\arrow["{1_{X\times Y}}", shift left, from=1-3, to=3-3]
	\arrow["{1_Y\circ p_Y}", from=1-3, to=3-5]
	\arrow["{p_X}"{description}, from=3-3, to=3-1]
	\arrow["{p_Y}"{description}, from=3-3, to=3-5]
\end{tikzcd}\]
    and by the universal property $1_X\times 1_Y = 1_{X\times Y}$.
\end{proof}
\begin{proposition}
    Let $1$ be the terminal object in $\catc$, then there exists an isomorphism
    \[\begin{tikzcd}
	{1\times X} && X
	\arrow["{\lambda_X}", from=1-1, to=1-3]
	\arrow["\simeq"', from=1-1, to=1-3]
    \end{tikzcd}\]
    that is natural in $X$. 
\end{proposition}
\begin{proof}[Proof]
    Let $\lambda_X = p_X$, then the following diagram commutes
    \[\begin{tikzcd}
	& {1\times X} \\
	1 && X \\
	& X
	\arrow[from=1-2, to=2-1]
	\arrow["{p_X}"{description}, from=1-2, to=2-3]
	\arrow["{\lambda_X}"{description}, from=1-2, to=3-2]
	\arrow[from=3-2, to=2-1]
	\arrow["{1_X}"', from=3-2, to=2-3]
    \end{tikzcd}\]
    We will now show that $(X,\bullet, p_X)$ is a product of $1$ and $X$ (by $\bullet$ we denote the unique morphism to the terminal object).
    Consider any $(U,\bullet, q_X)$ then $q_X$ is the unique morphism which makes the following diagram commute
    \[\begin{tikzcd}
	& U \\
	\\
	1 & X & X
	\arrow["{q_1}"{description}, curve={height=12pt}, from=1-2, to=3-1]
	\arrow["{q_X}"{description}, from=1-2, to=3-2]
	\arrow["{q_X}"{description}, curve={height=-12pt}, from=1-2, to=3-3]
	\arrow[from=3-2, to=3-1]
	\arrow["{1_X}"', from=3-2, to=3-3]
\end{tikzcd}\]
    therefore by lemma \ref{uniqueproductLemma} and the fact, that $\lambda_X$ is a morphism from $1\times X \to X$ which 
    commutes with projections we conclude that $\lambda_X$ is the unique isomorphism between $1\times X$ and $X$.\par
    To check naturality we have to prove that given $f: X \to Y$ the following diagram commutes
    \[\begin{tikzcd}
	{1\times X} && {1\times Y} \\
	\\
	X && Y
	\arrow["{\bullet\times f}"{description}, from=1-1, to=1-3]
	\arrow["{\lambda_X=p_X}"{description}, from=1-1, to=3-1]
	\arrow["{\lambda_Y=p_Y}"{description}, from=1-3, to=3-3]
	\arrow["f"{description}, from=3-1, to=3-3]
    \end{tikzcd}\]
    but $\bullet\times f$ is by definition the unique morphism such that the square commutes.
\end{proof}
\begin{remark}
    Analogously it can be shown that there is an isomorphism $\rho_X : X \times 1 \to X$, which is again
    given by $\rho_X = p_X$ and is natural in $X$.
\end{remark}
\begin{proposition}
    There exists an isomorphism 
    \[\begin{tikzcd}
	{(X\times Y)\times Z} && {X\times(Y\times Z)}
	\arrow["{\alpha_{X,Y,Z}}", from=1-1, to=1-3]
	\arrow["\simeq"', from=1-1, to=1-3]
    \end{tikzcd}\]
    natural in $X,Y$ and $Z$.
\end{proposition}
